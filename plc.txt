# 리스트값 입력 -> 짝수, 홀수 구분하기
num = [1,3,2,5,57,67,86,75,33,4,6,7,67,57,4,7,56,46]
result1 = []
result2 = []
for i in num:
    if i % 2 == 0:
        result1.append(i)
    else:
        result2.append(i)
print(result1)
print(result2)
